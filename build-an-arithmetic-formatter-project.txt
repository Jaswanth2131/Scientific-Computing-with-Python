def arithmetic_arranger(problems, show_answers=False):
    if len(problems) > 5:
        return "Error: Too many problems."

    first_line = ""
    second_line = ""
    dash_line = ""
    results_line = ""

    for index, problem in enumerate(problems):
        num1, operator, num2 = problem.split()

        if operator not in ["+", "-"]:
            return "Error: Operator must be '+' or '-'."

        if not num1.isdigit() or not num2.isdigit():
            return "Error: Numbers must only contain digits."

        if len(num1) > 4 or len(num2) > 4:
            return "Error: Numbers cannot be more than four digits."

        width = max(len(num1), len(num2)) + 2

        top = num1.rjust(width)
        bottom = operator + num2.rjust(width - 1)
        line = "-" * width

        if operator == "+":
            result = str(int(num1) + int(num2))
        else:
            result = str(int(num1) - int(num2))

        result = result.rjust(width)

        if index < len(problems) - 1:
            space = "    "
        else:
            space = ""

        first_line += top + space
        second_line += bottom + space
        dash_line += line + space
        results_line += result + space

    if show_answers:
        arranged = f"{first_line}\n{second_line}\n{dash_line}\n{results_line}"
    else:
        arranged = f"{first_line}\n{second_line}\n{dash_line}"

    return arranged
